програма створює папку і в ній десять папок. 
І в кожній з цих папок, ще по десять папок.
*****
string path = "D:\";
string folderName = "folder";
string str = "line";
string fulName;

CreateDir(path, folderName);

int i = 0;
int d = 0;
string s;
while (i < 10)
{
	CreateDir(path + folderName + "\", folderName + i);
	s = path + folderName + "\" + folderName + i;
	d = 0;
	while (d < 10)
	{
		CreateDir(s + "\", folderName + d);
		d = d + 1;
	}
	i = i + 1;
}
